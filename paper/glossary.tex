% Befehl für Eintrag, der sowohl im Glossar als auch im Abkürzungsverzeichnis auftaucht
\newcommand{\newglossaryacronym}[4]{
	\newglossaryentry{#1}{
		name=\glslink{#2}{#3},
		text=#3,
		description={#4},
		sort={#3}% nach Text sortieren
	}
	\newacronym[description={\glslink{#1}{#3}}]{#2}{#2}{#3}
}
